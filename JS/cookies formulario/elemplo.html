Exemple: Cookies

Creació de Cookie de un formulari - Pas 1

En el següent document web, rebem un nom del visitant i l'emmagatzemem a la galeta anomenada "Nom".

function createCookie(){

	var visitor_name =document.getElementById("name").value;
	var curdate = new Date();
	curdate.setMonth(curdate.getMonth() + 6);
	var cookie_date  = curdate.toUTCString();
	final_cookie = "Nom=" + encodeURIComponent(visitor_name) + "; expires= " + cookie_date+ ";path=/; domain=codepen.io";
	document.cookie = final_cookie;
	console.log(final_cookie);
	//console.log(document.cookie);
}
